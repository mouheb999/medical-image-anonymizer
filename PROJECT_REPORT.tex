\section*{Executive Summary}

This project implements a production-ready medical image anonymization system combining AI-powered image classification, dual OCR text detection, intelligent pixel redaction, and cloud storage integration. The system provides a complete web-based solution with FastAPI backend, modern frontend interface, and MinIO S3 storage for organized archival.

\begin{tcolorbox}[colback=gray!10,colframe=black]
\textbf{Key Achievement:} Successfully deployed a full-stack medical image anonymization platform with 7-stage processing pipeline, RESTful API, web interface, and category-organized S3 storage.
\end{tcolorbox}

% ---------- PIPELINE ----------
\section{7-Stage Processing Pipeline}

\subsection*{Stage 1 — Medical Image Classification}
\begin{itemize}[leftmargin=1.2cm]
    \item CLIP-based classification with domain + anatomy detection
    \item Categories: chest, dental, pelvic, skull, non\_medical, other\_medical
    \item Automatic rejection of non-medical images
    \item Confidence scoring for quality assurance
\end{itemize}

\subsection*{Stage 2 — Image Validation}
\begin{itemize}[leftmargin=1.2cm]
    \item Multi-format support: DICOM, JPEG, PNG, BMP, TIFF
    \item Validates required metadata and pixel data
    \item DICOM compression detection with clear error messages
    \item Graceful handling of unsupported formats
\end{itemize}

\subsection*{Stage 3 — Metadata Anonymization}
\begin{itemize}[leftmargin=1.2cm]
    \item Removes 12 PHI tags from DICOM files
    \item Preserves DICOM structure integrity
    \item Maintains diagnostic metadata
\end{itemize}

\subsection*{Stage 4 — Border Preprocessing}
\begin{itemize}[leftmargin=1.2cm]
    \item CLAHE enhancement on border regions only (15\% of image)
    \item Central diagnostic area preserved
    \item Improves OCR accuracy on edge text
    \item RGB conversion for grayscale images
\end{itemize}

\subsection*{Stage 5 — Dual OCR Detection}
\begin{itemize}[leftmargin=1.2cm]
    \item PaddleOCR + EasyOCR with parallel processing
    \item IoU-based deduplication (50\% threshold)
    \item 25\% higher detection rate than single engine
    \item Non-fatal EasyOCR fallback for robustness
\end{itemize}

\subsection*{Stage 6 — Pixel Redaction}
\begin{itemize}[leftmargin=1.2cm]
    \item OpenCV TELEA inpainting algorithm
    \item Border-only redaction (100px safety margin)
    \item Configurable padding (5px default)
    \item Preserves diagnostic regions
\end{itemize}

\subsection*{Stage 7 — Output Generation \& Storage}
\begin{itemize}[leftmargin=1.2cm]
    \item DICOM preserved via pydicom
    \item JPEG quality=95, PNG lossless
    \item Automatic mode conversion (RGBA→RGB, L→RGB for JPEG)
    \item MinIO S3 upload with category organization
    \item Presigned URL generation (24h expiry)
\end{itemize}

\subsection{MinIO Features}

\begin{itemize}[leftmargin=1.2cm]

% ---------- ERROR HANDLING ----------
\section{Error Handling \& Robustness}

\subsection{Implemented Safeguards}

\begin{itemize}[leftmargin=1.2cm]
    \item \textbf{Classification failure detection:} Stops pipeline if classification fails
    \item \textbf{DICOM compression handling:} Clear error messages for unsupported codecs
    \item \textbf{Pixel array validation:} None checks before processing
    \item \textbf{Image mode conversion:} Automatic RGBA→RGB, L→RGB for JPEG
    \item \textbf{Non-fatal EasyOCR:} Pipeline continues if EasyOCR fails
    \item \textbf{Non-fatal MinIO:} API succeeds even if storage upload fails
    \item \textbf{Comprehensive logging:} Debug prints at every stage
    \item \textbf{Timing metrics:} Performance monitoring per stage
    \item \textbf{Full traceback:} Detailed error reporting in JSON responses
\end{itemize}

\subsection{Dependency Compatibility}

\begin{tcolorbox}[colback=yellow!10,colframe=orange]
\textbf{Critical Constraint:} EasyOCR requires numpy<2.0, while pylibjpeg requires numpy>=2.0. Current configuration prioritizes EasyOCR compatibility. Compressed DICOM files (JPEG Lossless, JPEG 2000) are not supported.
\end{tcolorbox}

% ---------- DEPLOYMENT ----------
\section{Deployment}

\subsection{System Requirements}

\begin{itemize}[leftmargin=1.2cm]
    \item Python 3.11+
    \item 4GB+ RAM
    \item 2GB+ disk space (for models)
    \item MinIO server (Docker or standalone)
\end{itemize}

\subsection{Installation Steps}

\begin{enumerate}[leftmargin=1.2cm]
    \item Install dependencies: \texttt{pip install -r requirements.txt}
    \item Configure environment: Edit \texttt{.env} file
    \item Start MinIO: \texttt{docker compose up -d minio} or standalone
    \item Start API: \texttt{uvicorn api.main:app --reload --port 8000}
    \item Open frontend: \texttt{frontend/index.html} in browser
\end{enumerate}

\subsection{Configuration}

Environment variables in \texttt{.env}:

\begin{verbatim}
MINIO_ENDPOINT=localhost:9000
MINIO_ACCESS_KEY=minioadmin
MINIO_SECRET_KEY=minioadmin
MINIO_BUCKET=anonymized-images
MINIO_SECURE=false
OUTPUT_DIR=./output
TEMP_DIR=./api/temp
\end{verbatim}
